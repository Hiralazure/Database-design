categories [icon: category] {
  id serial pk
  name varchar(50)
  is_active BOOLEAN
}

products [icon: product] {
  id serail pk
  name varchar(150)
  category_id int fk
  description text
  price DECIMAL(10,2)
  color varchar(50)
  size varchar(50)
  product_type ENUM ('THRIFT', 'HANDMADE')
  quantity int
  condition VARCHAR(50)
  CHECK (
  (product_type = 'THRIFT' AND quantity = 1)
  OR
  (product_type = 'HANDMADE' AND quantity >= 0)
)
}

customers [icon: user] {
  id serail pk
  first_name varchar(50)
  last_name varchar(50)
  email varchar(322)
  phone_number varchar(150)
}
address [icon:address]{
  id serial pk
  customer_id int fk
  house_number varchar(50)
  address_2 varchar(250)
  city varchar(150)
  postcode varchar(150)
  country varchar(250)
}
order_status{
  id serial pk
  status varchar(150)
  description text
}
orders [icon: order] {
  id serial pk
  customer_id int fk
  billing_address_id int fk
  shipping_address_id int fk
  totalcost DECIMAL(10,2)
  status_id int fk
  tracking_id VARCHAR(100)
}
order_details{
  id serial pk
  order_id int fk
  product_id int fk
  quantity int
  price DECIMAL(10,2)
  color varchar(50)
  size varchar(50)
}
payments [icon: payment] {
  id PK
  order_id FK
  transaction_id VARCHAR
  amount DECIMAL(10,2)
  payment_method VARCHAR
  transaction_status VARCHAR
  payment_date TIMESTAMP
}

categories.id one-many products.category_id
products.id one-many orders.product_id
address.id one-one orders.shipping_address_id
address.id one-one orders.billing_address_id
customers.id one-many address.customer_id
customers.id one-many orders.customer_id
orders.status_id one-many order_status.id
orders.id one-manu order_details.order_id
orders.id one-one payments.order_id
